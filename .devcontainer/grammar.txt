# Backus-Naur Form Recap:
- "<>" are called chevrons, they denote *non-terminal* symbols, meaning that there will be another production rule to define its replacement
  - Ex. <fullname> ::= <title><name><name>
- text not surrounded by chevrons denote *terminal* symbols, they represent actual values that are allowed for the non-terminal symbol
- "|" the pipe symbol denotes alternatives
  - Ex. <title> ::= Mr|Mrs|Ms|Dr

# Lexemes
<digit> ::= 0::9
<alpha> ::= A::z

# Tokens
<num> ::= <digit> | <digit><num>
<var> ::= <alpha> | <alpha><var>

atom    : DIGIT | IDENTIFIER
        : LPAREN expr RPAREN

factor  : (PLUS | MINUS) atom

power   : factor (POW factor)*

term    : factor ((MUL | DIV) factor)*

expr    : term ((PLUS | MINUS) term)*

equa    : expr (EQUAL expr)*
